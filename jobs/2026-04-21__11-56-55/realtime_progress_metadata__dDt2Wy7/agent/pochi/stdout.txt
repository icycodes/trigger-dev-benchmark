import { task, wait, metadata } from "@trigger.dev/sdk/v3";

export const progressTask = task({
  id: "progressTask-realtime_progress_metadata__dDt2Wy7",
  run: async (payload: any) => {
    const totalItems = 10;
    
    for (let i = 1; i <= totalItems; i++) {
      const percentage = (i / totalItems) * 100;
      
      // Update the progress metadata
      metadata.set("progress", percentage);
      
      // Append a log entry
      metadata.append("logs", `Processed item ${i} of ${totalItems}`);
      
      console.log(`Processing item ${i}...`);
      
      // Simulate work
      await wait.for({ seconds: 1 });
    }
    
    return { completed: true };
  },
});
