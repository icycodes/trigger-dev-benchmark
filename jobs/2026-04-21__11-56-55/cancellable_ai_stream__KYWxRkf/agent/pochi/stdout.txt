import { task, streams, wait } from "trigger.dev";

export const aiStream = streams.define<string>({
  id: "aiStream-cancellable_ai_stream__KYWxRkf",
});

export const cancelStream = streams.input<{ reason?: string }>({
  id: "cancelStream-cancellable_ai_stream__KYWxRkf",
});

export const aiTask = task({
  id: "aiTask-cancellable_ai_stream__KYWxRkf",
  run: async (payload: any) => {
    let cancelled = false;
    
    cancelStream.on((message) => {
      console.log(`Cancelled: ${message.reason}`);
      cancelled = true;
    });

    for (let i = 0; i < 10; i++) {
      if (cancelled) {
        break;
      }
      
      await aiStream.append(`Chunk ${i + 1}`);
      await wait.for({ seconds: 1 });
    }
  }
});
