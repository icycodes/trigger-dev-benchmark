import { task, batch, wait } from "@trigger.dev/sdk/v3";

export const childTask = task({
  id: "childTask-parallel_batch_trigger__KEHonCM",
  run: async (payload: number) => {
    await wait.for({ seconds: 2 });
    return payload * payload;
  },
});

export const parentTask = task({
  id: "parentTask-parallel_batch_trigger__KEHonCM",
  run: async (payload: number[]) => {
    const result = await batch.triggerByTaskAndWait(
      payload.map((num) => ({
        task: childTask,
        payload: num,
      }))
    );

    return result.runs.map((run) => {
      if (run.ok) {
        return run.output;
      } else {
        throw new Error(`Task failed: ${run.error}`);
      }
    });
  },
});
