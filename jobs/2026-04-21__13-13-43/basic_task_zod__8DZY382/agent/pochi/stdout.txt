import { task } from "@trigger.dev/sdk/v3";
import { z } from "zod";

export const basicZodTask = task({
  id: "basic-zod-task-basic_task_zod__8DZY382",
  schema: z.object({
    name: z.string(),
    age: z.number(),
  }),
  retry: {
    maxAttempts: 3,
  },
  run: async (payload) => {
    return `Hello ${payload.name}, you are ${payload.age} years old!`;
  },
});
