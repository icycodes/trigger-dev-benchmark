```typescript
import { task, wait } from "@trigger.dev/sdk/v3";

export const delayedNotify = task({
  id: "delayed-notify-${trial_id}",
  run: async (payload: { userId: string; message: string }) => {
    // Wait for 10 seconds using Trigger.dev's durable wait
    await wait.for({ seconds: 10 });

    // After the wait, return the result
    return {
      sent: true,
      message: payload.message,
    };
  },
});
```

The implementation follows the Trigger.dev v3 SDK patterns, ensuring that the task execution is checkpointed during the wait period to save compute resources.
